@name AFK_Mass
@persist MASS_MAX MASS_MUL
@persist State [E EC]:entity

#UNUSED
#@persist MASS_OTHER

interval(1000)

if(first())
{
    MASS_MAX = 50000
    MASS_MUL = 2000

    # MASS_OTHER = MASS_MAX / MASS_MUL

    E = entity()
    EC = E:isConstrainedTo()
}

if(State)
{
    EC:setMass(MASS_MAX)
}
else
{
    EC:setMass(MASS_MUL)
}
